#include <WiFi.h>
#include <HTTPClient.h>
#include <DHT.h>
#define DHT22_PIN  21
const char* ssid = "MidzaMidza"; //Promjena wifija na hotspot teleofona, globalsoft ima problem sa spajanjem
const char* password = "midza432001";
const char* serverUrl = "http://192.168.43.153:8080/post-data";
DHT dht22(DHT22_PIN, DHT22);
void setup() {
  Serial.begin(115200);
  dht22.begin();
  WiFi.begin(ssid, password);

  while (WiFi.status() != WL_CONNECTED) {
    delay(1000);
    Serial.println("Connecting...");
  }
  Serial.println("Connected to WiFi");
}

void loop() {
  if (WiFi.status() == WL_CONNECTED) {
    HTTPClient http;
    http.begin(serverUrl);
    float hum = dht22.readHumidity();
    float temp = dht22.readTemperature();
        
        if (isnan(hum) || isnan(temp)) {
            Serial.println("Greška pri očitanju!");
            return;
        }
    http.setTimeout(5000); 
    String postData = "temp=" + String(temp) + "hum="+ String(hum);

    http.addHeader("Content-Type", "application/x-www-form-urlencoded");
    int httpResponseCode = http.POST(postData);
    if (httpResponseCode > 0) {
      String response = http.getString();
      Serial.println(response);
    } else {
      Serial.print("Error on sending POST: ");
      Serial.println(httpResponseCode);
    }
    http.end();
  }
  delay(1000);  
}
